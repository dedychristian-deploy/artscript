sub GetDataFile()
		
	if arrRow.ubound = -1 then exit sub

		if len(arrRow[i]) > 0 then

end sub

Sub RunningKelompok1PertamaKali()
	dim Kelompok1_lc as container =  Scene.FindContainer("KELOMPOK1")
	dim Kelompok2_lc as container =  Scene.FindContainer("KELOMPOK2")
	
	'reset kelompok 2
	Kelompok2_lc.Script.p_start		
	Kelompok2_lc.Script.p_pause
	Kelompok2_lc.Script.p_reset true

	'isi beritanya
	counter_li = 0
	scene.findcontainer("ISI_KONTEN_NEWSTICKER1").geometry.text = arrRow2[counter_li]
	counter_li = counter_li + 1 
	scene.findcontainer("ISI_KONTEN_NEWSTICKER2").geometry.text = arrRow2[counter_li] 
	'counter_li = -1
	
	'jalankan kelompoknya
	Kelompok1_lc.Script.p_pause
	Kelompok1_lc.Script.p_reset true
	Kelompok1_lc.Script.p_start		
	
	'nyalakan kembali burung kelompok 1 baris 2 dan burung kelompok 2 baris 2, 
	'untuk jaga2 karena mungkin masih tidak aktif saat berita terakhir bila ada iklan
	scene.findcontainer("WITH_IMG2").Active = true
	scene.findcontainer("WITH_IMG4").Active = true
end sub
