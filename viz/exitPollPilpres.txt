sub StabilitasSuaraDown()
	
	'jumlah kandidat
	jmlkandidat_li = GetParameterInt("StabilitasSuara_Jumlah_Kandidat")
	indexTotalSuara_li = GetParameterInt("StabilitasSuara_Index_TotalSuara")
	indexTPS_li = GetParameterInt("StabilitasSuara_Index_TPS")
	indexPartisipasi_li = GetParameterInt("StabilitasSuara_Index_Partisipasi")
	
	'lokasi file line graphic
	alamatfile = GetParameterString("Root_Path") & GetParameterString("StabilitasSuara")		
	'println "alamatfile : " & alamatfile

	'ambil isi file
	System.LoadTextFile(alamatfile, isifile)		
	println "isifile -> " & isifile

	'ambil data pisahkan dengan baris baru
	isifile.split(Chr(10), IsiKandidat)	

	'cek baris kosong		
	for j = 0 to IsiKandidat.ubound - 1
		if IsiKandidat[j] <> "" then totalbaris_li = j
	next
	'println "(total index) = " & totalbaris_li
	SendCommand("RENDERER*FUNCTION*DataPool*Data SET JUDUL = STABILITAS SUARA")

	'ambil data terakhir
	IsiKandidat[totalbaris_li].split(";", strISIpersen)	
	'kolom index ke 4 untuk total suara masuk
	scene.FindContainer("datamasuk").geometry.text = strISIpersen[indexTotalSuara_li] & " %"
	
	''cmd = "RENDERER*FUNCTION*DataPool*Data SET DATA_MASUK = " & strISIpersen[indexTotalSuara_li]
	''SendCommand(cmd)

	
	'---------------------------------
	'untuk nilai jika dibawah 10 baris
	if IsiKandidat.ubound <= 6 then
		'println "IsiKandidat.ubound : " & IsiKandidat.ubound
			
			'jika datanya masih kosong, maka reset semua nilai line ke 0
			if 	IsiKandidat[0] = "" then
				'println "dibawah 6 tapi kosong"
				'9 itu jmlh baris plugin GraphLine yg digunakan
				for semuabaris = 0 to 5
					
					'lakukan sebanyak jumlah kandidat kurang 1 (hitung index)
					for i = 0 to jmlkandidat_li - 1
						send = "RENDERER*TREE*$Graph" & i & "*GEOM*X" & semuabaris & " SET 0"
						sendcommand(send)
						'println send
						send = "RENDERER*TREE*$Graph" & i & "*GEOM*Y" & semuabaris & " SET 0"
						sendcommand(send)
						'println send
					next
					
					scene.FindContainer("TPS-" & semuabaris).geometry.text = Cstr(semuabaris + 1)
					'println "TPS-" & semuabaris & " = " & Cstr(semuabaris + 1)
					'scene.FindContainer("datamasuk").geometry.text = "0.00 %"
					
					''cmd = "RENDERER*FUNCTION*DataPool*Data SET DATA_MASUK = 0"
					''SendCommand(cmd)
				next
				
			else
					
				'jika datanya yang masuk masih dibawah 5
				kelipatansepuluhawal = 0
				'println "dibawah 6 tapi isi"
				'println "IsiKandidat.ubound : " & IsiKandidat.ubound
				for	semuabarisawal = 0 to totalbaris_li
					IsiKandidat[semuabarisawal].split(";", strISI)
					'println "jmlkandidat_li : " & jmlkandidat_li
					
					'isi nilai suara perkandidat
					'lakukan sebanyak jumlah kandidat kurang 1 (hitung index)
					for i = 0 to jmlkandidat_li - 1
						send = "RENDERER*TREE*$Graph" & i & "*GEOM*Y" & semuabarisawal & " SET " & strISI[i]
						sendcommand(send)
						'println send
						
						'isi jarak tiap 10 titik
						send = "RENDERER*TREE*$Graph" & i & "*GEOM*X" & semuabarisawal & " SET " & kelipatansepuluhawal
						sendcommand(send)
						'println send

					next
					
					'isi nilai tps masuk perkandidat nilai tiap 10 titik mengikuti nilai terakhir tiap baris
					'scene.FindContainer("nilai1").geometry.text = Cstr(0)
					scene.FindContainer("TPS-" & semuabarisawal).geometry.text = strISI[indexTPS_li]
					'println "TPS-" & semuabarisawal & " = " & strISI[indexTPS_li]

					kelipatansepuluhawal = kelipatansepuluhawal + 18
					
					'println "kelipatansepuluhawal -> " & kelipatansepuluhawal
					if 	kelipatansepuluhawal >= 91 then
						kelipatansepuluhawal = 0
					end if
				next
			end if
	end if

	if IsiKandidat.ubound > 6 then
		'println "diatas 6"
		kelipatansepuluh = 0
		tiapkandidat = 0
		'println "IsiKandidat.ubound ---------- 1 : " & IsiKandidat[IsiKandidat.ubound - 1]
		
		'untuk ambil 5 nilai terakhir
		for semuabaris = totalbaris_li - 5 to totalbaris_li
			'println "baris yang diambil = " & IsiKandidat[semuabaris]
			IsiKandidat[semuabaris].split(";", strISI)
						
			'isi keseluruhan tiap baris			
			for i = 0 to jmlkandidat_li - 1
				send = "RENDERER*TREE*$Graph" & i & "*GEOM*Y" & tiapkandidat & " SET " & strISI[i]
				sendcommand(send)
				'println send
				'isi jarak tiap 5 titik
				send = "RENDERER*TREE*$Graph" & i & "*GEOM*X" & tiapkandidat & " SET " & kelipatansepuluh 
				sendcommand(send)
				'println send
			next
				
			'isi nilai tps masuk perkandidat nilai tiap 10 titik mengikuti nilai terakhir tiap baris
			scene.FindContainer("TPS-" & tiapkandidat).geometry.text = strISI[indexTPS_li]
			'println "TPS-" & tiapkandidat & " = " & strISI[indexTPS_li]
			
			'println "tiapkandidat -> " & tiapkandidat
			
			tiapkandidat = tiapkandidat + 1
			if tiapkandidat > 6 then
				tiapkandidat = 0
			end if
				
			kelipatansepuluh = kelipatansepuluh + 18
			if kelipatansepuluh >= 91 then
				kelipatansepuluh = 0
			end if
		next
	end if
end sub

Function ConvertZona(zona as string) as string
	if zona.Find("SUMATERA") >= 0 then
		ConvertZona = "1"
	elseif zona.Find("DKI+BANTEN") >= 0 then
		ConvertZona = "2"
	elseif zona.Find("JABAR") >= 0 then
		ConvertZona = "3"				
	elseif zona.Find("JATENG+DIY") >= 0 then
		ConvertZona = "4"	
	elseif zona.Find("JATIM") >= 0 then
		ConvertZona = "5"
	elseif zona.Find("BALI") >= 0 then
		ConvertZona = "6"		
	elseif zona.Find("KALIMANTAN") >= 0 then
		ConvertZona = "7"		
	elseif zona.Find("SULAWESI") >= 0 then
		ConvertZona = "8"		
	elseif zona.Find("LAINNYA") >= 0 then
		ConvertZona = "9"			
	end if
End Function

Sub ShowGraphic(pilihangraphic as integer, propinsi as string)

	for i = 1 to 33
		Scene.FindContainer("GRAPHIC_" & i).Active = False
		SendCommand("RENDERER*STAGE*DIRECTOR*GRAPHIC_" & i & " SHOW 0")
	next
	
	if pilihangraphic = 0 then
		'iklan
		Scene.FindContainer("CREATIVE-PLACEMENT").Active = True	
	else
		'Scene.FindContainer("CREATIVE-PLACEMENT").Active = False	
		SendCommand("RENDERER*STAGE*DIRECTOR*CP-LAYAR-TENGAH-3 SHOW 0")
		SendCommand("RENDERER*STAGE*DIRECTOR*CP-LAYAR-TENGAH-3 CONTINUE")			
	end if	

	Scene.FindContainer("GRAPHIC_" & pilihangraphic).Active = true
	SendCommand("RENDERER*STAGE*DIRECTOR*GRAPHIC_" & pilihangraphic & " SHOW 0")
	SendCommand("RENDERER*STAGE*DIRECTOR*GRAPHIC_" & pilihangraphic & " CONTINUE")	

	'capres kanan kiri
	if pilihangraphic = 1 or pilihangraphic = 30 then

		SendCommand("RENDERER*STAGE*DIRECTOR*3DMODEL-CAPRES CONTINUE")

		Sebaran_Perolehan_Suara(GetParameterString("PerolehanSuara"))
