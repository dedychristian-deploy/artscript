Sub isiNama(katakunci as string, isi as string)
	if katakunci.Find("nama") >= 0 then
		
		Scene.FindContainer(katakunci).Geometry.Text = isi
		
	end if
	
	if katakunci.Find("account") >= 0 then
		
		Scene.FindContainer(katakunci).Geometry.Text = isi
		
	end if
	
end sub

Sub isiAbout(katakunci as string, isi as string)
	if katakunci.Find("about") >= 0 then
		
		Scene.FindContainer("KONTEN-ABOUT").Geometry.Text = isi

	else
		
	end if
end sub

Sub isiPost(katakunci as string, isi as string)
	if katakunci.Find("tweets") >= 0 then
		
		Scene.FindContainer(katakunci).Geometry.Text = isi
		
		println isi

	end if
end sub
