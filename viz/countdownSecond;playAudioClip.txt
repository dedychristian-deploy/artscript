Sub Peringatan()

	if Digit(CStr(GetParameterInt("PeringatanSatu_tm"))) = Scene.FindContainer("nilai").Geometry.Text then
		cmd= "RENDERER*TREE*$PeringatanSatu*AUDIO*TEST_CLIP PLAY" 
		SendCommand(cmd)
		println"tttteeenggg1"
	
	elseif Digit(CStr(GetParameterInt("PeringatanDua_tm"))) = Scene.FindContainer("nilai").Geometry.Text then
		cmd= "RENDERER*TREE*$PeringatanDua*AUDIO*TEST_CLIP PLAY" 
		SendCommand(cmd)
		println "dua"
	
	elseif Digit(CStr(GetParameterInt("PeringatanTiga_tm"))) >= Scene.FindContainer("nilai").Geometry.Text and Scene.FindContainer("nilai").Geometry.Text > "00" then
		cmd= "RENDERER*TREE*$PeringatanTiga*AUDIO*TEST_CLIP PLAY" 
		SendCommand(cmd)
		println "tiga"
		
	elseif COUNTER = 0 and  Scene.FindContainer("nilai").Geometry.Text = "00" then
		cmd= "RENDERER*TREE*$PeringatanAkhir*AUDIO*TEST_CLIP PLAY" 
		SendCommand(cmd)
		println "lima"
	

	
	end if
	
	
	
End Sub

Function Digit(timer as string) as string
	if timer.length = 1 then
		timer = "0" & timer
	end if 
	Digit = timer
end function
