sub Isi()

	'ambil isi file text
	lokasifile = GetParameterString("LokasiText")
	System.LoadTextFile(lokasifile, strlokasifile)
	
	strlokasifile.split(Chr(10), arrstrlokasifile)
	
	for i = 0 to arrstrlokasifile.ubound
		arrstrlokasifile[i].split(";", arrstrisibaris) 
		CekNilai(arrstrisibaris[0], arrstrisibaris[1])
				
	next
	
	
	
end sub

Sub CekNilai(katakunci as string, nilai as string)
	'### MEMASUKAN DATA BERDASARKAN KATA KUNCI 
	Scene.FindContainer(katakunci).Geometry.Text = nilai
	if katakunci = "PATH_IMAGE" then
	cmd = "RENDERER*TREE*$PATH_IMAGE*FUNCTION*ControlImage*input SET " & nilai
	SendCommand(cmd)
	'#643630*FUNCTION*ControlImage*image_prefix SET ASDF.JPG
	end if
	
end sub
