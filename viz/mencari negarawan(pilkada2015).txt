sub Isi()

	'ambil isi file text
	'lokasifile = GetParameterString("LokasiText")

	strlokasifile.split(Chr(10), arrstrlokasifile)

		arrstrlokasifile[i].split(";", arrstrisibaris)

Sub SetImage(katakunci as string, nilai as string)
	'### MEMASUKAN DATA BERDASARKAN KATA KUNCI 
	for i = 1 to 2
	
	if katakunci = "path_image"&i then
		cmd = "RENDERER*TREE*$FOTO_"& i &"*FUNCTION*ControlImage*input SET " & nilai
		SendCommand(cmd)
	end if
	
	next
end sub
