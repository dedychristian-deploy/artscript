sub icon(namaicon as string,pathfile as string)

	reseticon()
	
	'baca isi file kumpulan alamat file
	lokasifile = GetParameterString("Path_File") & pathfile & ".txt"
	System.LoadTextFile(lokasifile, strlokasifile)
	'println "lokasifile : " & lokasifile
	
	'pisahkan alamat file perbaris kedalam array
	strlokasifile.split(Chr(10), arrisibarisdata)
	'println "strlokasifile : " & strlokasifile
	
	
	for i = 1 to arrisibarisdata.ubound - 2
		arrisibarisdata[i].split(";", arrstrisibaris) 
		
		
		scene.findcontainer("LAPORAN_"&i).Geometry.text =  arrstrisibaris[0]
		scene.findcontainer("SUB_LAPORAN_"&i).Geometry.text =  arrstrisibaris[1]


	next
	
		
	for i = 0 to arrisibarisdata.ubound - 2
		arrisibarisdata[0].split(";", arrstrisibaris) 
		
		scene.findcontainer("NAMA_KOTA").Geometry.text =  arrstrisibaris[0]


	next
	
	
	
	
	
	send = "RENDERER*TREE*$" & namaicon & "*ACTIVE SET on"
	SendCommand(send)
	
	
	send = "RENDERER*STAGE*DIRECTOR*INKONTEN SHOW 0"
	SendCommand(send)
	send = "RENDERER*STAGE*DIRECTOR*INKONTEN CONTINUE"
	SendCommand(send)
	
	'println pathfile

end sub
