sub ceknamaprogram()
namaProgram = System.map["NAMA_PROGRAM"]
if namaProgram = "METRO PAGI" then
	metropagi()
	Scene.FindContainer("JUDULNYA").geometry.text = namaProgram
elseif namaProgram = "METRO KINI" then
	metrokini()
	Scene.FindContainer("JUDULNYA").geometry.text = namaProgram
	
elseif namaProgram = "METRO SIANG" then
	metrosiang()
	Scene.FindContainer("JUDULNYA").geometry.text = namaProgram
	
elseif namaProgram = "METRO SORE" then
	metrosore()
	Scene.FindContainer("JUDULNYA").geometry.text = namaProgram
	
		
elseif namaProgram = "METRO HARI INI" then
	metrohariini()
	Scene.FindContainer("JUDULNYA").geometry.text = namaProgram
	
elseif namaProgram = "TOP NEWS" then
	topnews()
	Scene.FindContainer("JUDULNYA").geometry.text = namaProgram
	
elseif namaProgram = "METRO MALAM" then
	metromalam()
	Scene.FindContainer("JUDULNYA").geometry.text = namaProgram
	
	else
	Scene.FindContainer("JUDULNYA").geometry.text = namaProgram
end if
end sub
