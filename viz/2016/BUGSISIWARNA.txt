dim toggle as boolean 
dim delay as boolean
dim inc as integer

sub OnInitParameters()
	RegisterParameterContainer("_controledParentLiveCont", "Parent Live Control Container")
	RegisterParameterString( "_directorToRun","Director To Play", "PROGRAM", 40, 100, "" )
	RegisterParameterInt( "_delay","Delay on Change", 20 , 0 , 1000)
end sub

sub OnLButtonDown()
end sub

sub OnLButtonUp() 
	if not(toggle) then
	
		System.map["NAMA_PROGRAM"] = Scene.FindContainer("TEKS_PROGRAM").Geometry.Text
		GantiBackgroundProgram()
	
		toggle = true		 	
		
		SendCommand("RENDERER*FRONT_LAYER*FUNCTION*DataPool*Data SET BUGS_JUDUL_PROGRAM = 1") 
		SendCommand("RENDERER*FUNCTION*DataPool*Data SET BUGS_JUDUL_PROGRAM = 1")
		
		scene.findcontainer("$BUNDEL_LOGO$LOGO_PROGRAM").active = toggle	
		stage.finddirector( GetparameterString("_directorToRun") ).reverse = false
		stage.finddirector( GetparameterString("_directorToRun") ).ContinueAnimation()
		stage.finddirector("INOUT-BUGS-GLOBE") .reverse = false
		stage.finddirector("INOUT-BUGS-GLOBE") .ContinueAnimation()
		
		GetParameterContainer("_controledParentLiveCont").script.setprogram_onair(toggle)
		this.childcontainer.material.color.rgb = cvertex(255,0,0)/255
	else
		toggle = false		
		SendCommand("RENDERER*FRONT_LAYER*FUNCTION*DataPool*Data SET BUGS_JUDUL_PROGRAM = 0") 
		SendCommand("RENDERER*FUNCTION*DataPool*Data SET BUGS_JUDUL_PROGRAM = 0")
		stage.finddirector( GetparameterString("_directorToRun") ).reverse = true
		stage.finddirector( GetparameterString("_directorToRun") ).ContinueAnimation()
		stage.finddirector("INOUT-BUGS-GLOBE").reverse = true
		stage.finddirector("INOUT-BUGS-GLOBE").ContinueAnimation()
		GetParameterContainer("_controledParentLiveCont").script.setprogram_onair(toggle)		
		delay = true
	end if
	
end sub

sub OnInit()
	delay = false
	toggle = false
	inc = 0
	stage.finddirector( GetParameterString("_directorToRun") ).time = 0
	stage.finddirector( GetParameterString("_directorToRun") ).reverse = false
	this.childcontainer.material.color.rgb = cvertex(0,0,0)/255
	scene.findcontainer("$BUNDEL_LOGO$LOGO_PROGRAM").active = false
	GetParameterContainer("_controledParentLiveCont").script.setprogram_onair(toggle)	
end sub

sub OnExecPerField()
	if delay then
		inc++
		if inc >= GetparameterInt( "_delay" ) then
			scene.findcontainer("$BUNDEL_LOGO$LOGO_PROGRAM").active = toggle	
			this.childcontainer.material.color.rgb = cvertex(0,0,0)/255
			inc = 0
			delay = false
			'println is_onairFrom
			'if il_onair and this.name <> is_onairFrom then
			'	OnLButtonUp()
			'end if	
		end if
		
	end if
end sub
